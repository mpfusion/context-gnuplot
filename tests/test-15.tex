	[gnuplot]

%% Test TeX point sets

	[context]
	[points=tex,
	 pointset=myset]

\defineconversion
	[myset]
	[-,{\ss x},\mathematics{\circ}]

\startGNUPLOTscript [myfunction]
	plot 'somedata' using 1:1 with points, \
		'' using 1:2 with points pointtype 2
\stopGNUPLOTscript

\useGNUPLOTgraphic [myfunction]

\stoptext
